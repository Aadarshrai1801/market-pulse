# --- OCR endpoint (/api/ocr) ---
opencv-python-headless==4.11.0.86   # headless build - no GUI deps needed on a server

# PINNED <2.0: opencv-python-headless and paddlepaddle==2.6.2 both ship
# binaries compiled against numpy 1.x's C ABI. numpy 2.0 changed that ABI,
# so numpy>=2 causes "numpy.core.multiarray failed to import" / "module
# compiled against ABI version 0x1000009 but this version of numpy is
# 0x2000000" at gunicorn boot. Confirmed working locally at 1.26.4.
numpy==1.26.4
pandas==2.2.2

# PaddleOCR - the OCR engine actually used by ocr/ocr.py.
# CPU-only build (no GPU on Render/Fly free tiers) - do NOT switch to
# paddlepaddle-gpu, that requires CUDA and will fail to install/run here.
#
# PINNED exactly: ocr.py calls PaddleOCR(use_gpu=..., show_log=...), which
# are old-style constructor args removed in PaddleOCR 3.x. paddleocr>=2.7
# lets pip grab the latest 3.x release, which crashes at import time with
# "ValueError: Unknown argument: use_gpu". Stay on this exact 2.x pair
# until ocr.py is updated to the new API.
paddlepaddle==2.6.2
paddleocr==2.7.3

# --- Production server ---
gunicorn==26.0.0           # run with: gunicorn -w 1 -b 0.0.0.0:$PORT app:app
                           # NOTE: -w 1, not -w 2 - see "Single worker" note below
                           # (Windows/dev alternative: waitress)
